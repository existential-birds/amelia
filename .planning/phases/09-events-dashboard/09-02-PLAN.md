```typescript

pr_number: number | null;
pr_title: string | null;
pr_comment_count: number | null;
pr_comments: PRCommentData[] | null;  // Only present for pr_auto_fix workflows
```

```typescript
interface PRCommentData {
  comment_id: number;
  file_path: string | null;
  line: number | null;
  body: string;  // Truncated to 200 chars
  author: string;
  status: 'fixed' | 'failed' | 'skipped';
  status_reason: string | null;
  html_url: string;  // GitHub comment URL
}
```
